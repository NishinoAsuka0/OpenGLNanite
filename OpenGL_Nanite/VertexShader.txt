#version 330 core
layout (location = 0) in vec3 position;
layout(location = 1) in vec3 color;				// ัีษซฑไมฟตฤถฅตใส๔ะิฮปึรึตฮช 1
out vec3 ourColor;
uniform mat4 transformMatrix;						
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;
void main()
{
	ourColor = color;
	gl_Position = projectionMatrix * viewMatrix * transformMatrix * vec4(position, 1.0f);	 // ืขาโพุี๓ณหทจาชดำำาฯ๒ื๓ถม
}
